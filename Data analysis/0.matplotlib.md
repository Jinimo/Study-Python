import matplotlib.pyplot as plt
import platform

# 시스템별 폰트 적용

elif platform.system() == "Windows":       # Windows 시스템
    plt.rc("font", family = "Malgun Gothic")
elif platform.system() == "Linux":         # Linux 시스템 (코랩)
    plt.rc("font", family = "Malgun Gothic")
